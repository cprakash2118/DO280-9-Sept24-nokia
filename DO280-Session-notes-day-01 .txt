 - Imperative Workflows

 - Declarative Workflows:-
 
apiVersion: v1
kind: Pod
metadata:
  name: example-pod
  labels:
    app: myapp
spec:
  containers:
  - name: myapp-container
    image: httpd 
    imagePullPolicy: IfNotPresent
    name: httpd
    ports:
    - containerPort: 8080
      protocol: TCP
